var count = 0;
var originalimage = "url(" + jQuery('.accordion').data('background') + ")";
jQuery('.accordion').css('background-image', originalimage);
jQuery('.accordion-item').each(function() {
    jQuery('<img/>')[0].src = jQuery(this).data('background');
})
jQuery('.accordion .accordion-item').on('click', function() {
    var clicked = jQuery(this);
    count++;
    var is_click = jQuery(this).data('isclick');
    var is_click_pass = "";
    // console.log("count",count);
    var bg = "url(" + jQuery(this).data('background') + ")";
    clicked.parent('.accordion').css('background-image', bg).find('.accordion-item').not(clicked).removeClass('active');
    if (is_click == "1") {
        jQuery(this).attr('data-isclick', '0');
        jQuery('.accordion').css('background-image', originalimage);
    } else {
        jQuery(this).attr('data-isclick', '1');
    }
    clicked.addClass('active');
    jQuery('.accordion').addClass('active');
    var is_click = jQuery(this).data('isclick');
    if (jQuery(this).hasClass("active")) {
        if (is_click == "1") {
            is_click_pass = "1";
            // console.log("is actiev");
            jQuery(this).attr('data-isclick', '1');
            setTimeout(function() {
                var data_CutImg = jQuery(".accordion .active").data('background');
                // console.log("data_CutImg", data_CutImg);
                jQuery('.accordion').css('background-image', 'url(' + data_CutImg + ')');
            }, 100);
        } else {
            // console.log("is  not actiev");
            jQuery(this).attr('data-isclick', '0');
            // console.log("originalimage", originalimage);
        }
    }
});
